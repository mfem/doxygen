13 // Parallel Low-Order Refined Solvers Miniapp
14 // ------------------------------------------
15 //
16 // This miniapp illustrates the use of low-order refined preconditioners for
17 // finite element problems defined using H1, H(curl), H(div), or L2 finite
18 // element spaces. The following problems are solved, depending on the chosen
19 // finite element space:
20 //
21 // H1 and L2: definite Helmholtz problem, u - Delta u = f
22 // (in L2 discretized using the symmetric interior penalty DG method)
23 //
24 // H(curl): definite Maxwell problem, u + curl curl u = f
25 //
26 // H(div): grad-div problem, u - grad(div u) = f
27 //
28 // In each case, the high-order finite element problem is preconditioned using a
29 // low-order finite element discretization defined on a Gauss-Lobatto refined
30 // mesh. The low-order problem is solved using hypre's AMG preconditioners:
31 // BoomerAMG is used for H1 and L2 problems, AMS is used for H(curl) and 2D
32 // H(div) problems, and ADS is used for 3D H(div) problems.
33 //
34 // For vector finite element spaces, the special "Integrated" basis type is used
35 // to obtain spectral equivalence between the high-order and low-order refined
36 // discretizations. This basis is defined in reference [1] and spectral
37 // equivalence is shown in [2]:
38 //
39 // [1]. M. Gerritsma. Edge functions for spectral element methods. Spectral and
40 // High Order Methods for Partial Differential Equations. (2010)
41 // [2]. C. Dohrmann. Spectral equivalence properties of higher-order tensor
42 // product finite elements and applications to preconditioning. (2021)
43 //
44 // The action of the high-order operator is computed using MFEM's partial
45 // assembly/matrix-free algorithms (except in the case of L2, which remains
46 // future work).
47 //
48 // Compile with: make plor_solvers
49 //
50 // Sample runs:
51 //
52 // mpirun -np 4 plor_solvers -m ../../data/fichera.mesh -fe h
53 // mpirun -np 4 plor_solvers -m ../../data/fichera.mesh -fe n
54 // mpirun -np 4 plor_solvers -m ../../data/fichera.mesh -fe r
55 // mpirun -np 4 plor_solvers -m ../../data/fichera.mesh -fe l
56 // mpirun -np 4 plor_solvers -m ../../data/amr-hex.mesh -fe h -rs 0 -o 2
57 // mpirun -np 4 plor_solvers -m ../../data/amr-hex.mesh -fe l -rs 0 -o 2
58 //
59 // Device sample runs:
60 // mpirun -np 4 plor_solvers -m ../../data/fichera.mesh -fe h -d cuda
61 // mpirun -np 4 plor_solvers -m ../../data/fichera.mesh -fe n -d cuda
62 // mpirun -np 4 plor_solvers -m ../../data/fichera.mesh -fe r -d cuda
63 // mpirun -np 4 plor_solvers -m ../../data/fichera.mesh -fe l -d cuda

91  args.AddOption(&ser_ref_levels, "-rs", "--refine-serial",
92  "Number of times to refine the mesh uniformly in serial.");
93  args.AddOption(&par_ref_levels, "-rp", "--refine-parallel",
94  "Number of times to refine the mesh uniformly in parallel.");
95  args.AddOption(&order, "-o", "--order", "Polynomial degree.");
96  args.AddOption(&fe, "-fe", "--fe-type",
97  "FE type. h for H1, n for Hcurl, r for Hdiv, l for L2");
98  args.AddOption(&visualization, "-vis", "--visualization", "-no-vis",
99  "--no-visualization",
100  "Enable or disable GLVis visualization.");
101  args.AddOption(&device_config, "-d", "--device",
102  "Device configuration string, see Device::Configure().");
103  args.ParseCheck();
104 
105  Device device(device_config);
106  if (Mpi::Root()) { device.Print(); }
107 
108  bool H1 = false, ND = false, RT = false, L2 = false;
109  if (string(fe) == "h") { H1 = true; }
110  else if (string(fe) == "n") { ND = true; }
111  else if (string(fe) == "r") { RT = true; }
112  else if (string(fe) == "l") { L2 = true; }
113  else { MFEM_ABORT("Bad FE type. Must be 'h', 'n', 'r', or 'l'."); }
114 
115  if (RT) { grad_div_problem = true; }
116  double kappa = (order+1)*(order+1); // Penalty used for DG discretizations
117 
118  Mesh serial_mesh(mesh_file, 1, 1);
119  int dim = serial_mesh.Dimension();
120  MFEM_VERIFY(dim == 2 || dim == 3, "Spatial dimension must be 2 or 3.");
121  for (int l = 0; l < ser_ref_levels; l++) { serial_mesh.UniformRefinement(); }
122  ParMesh mesh(MPI_COMM_WORLD, serial_mesh);
123  for (int l = 0; l < par_ref_levels; l++) { mesh.UniformRefinement(); }
124  serial_mesh.Clear();
125 
126  if (mesh.ncmesh && (RT || ND))
127  { MFEM_ABORT("LOR AMS and ADS solvers are not supported with AMR meshes."); }
128 
129  FunctionCoefficient f_coeff(f), u_coeff(u);
130  VectorFunctionCoefficient f_vec_coeff(dim, f_vec), u_vec_coeff(dim, u_vec);
131 
132  int b1 = BasisType::GaussLobatto, b2 = BasisType::IntegratedGLL;
133  unique_ptr<FiniteElementCollection> fec;
134  if (H1) { fec.reset(new H1_FECollection(order, dim, b1)); }
135  else if (ND) { fec.reset(new ND_FECollection(order, dim, b1, b2)); }
136  else if (RT) { fec.reset(new RT_FECollection(order-1, dim, b1, b2)); }
137  else { fec.reset(new L2_FECollection(order, dim, b1)); }
138 
139  ParFiniteElementSpace fes(&mesh, fec.get());
140  HYPRE_Int ndofs = fes.GlobalTrueVSize();
141  if (Mpi::Root()) { cout << "Number of DOFs: " << ndofs << endl; }
142 
143  Array<int> ess_dofs;
144  // In DG, boundary conditions are enforced weakly, so no essential DOFs.
145  if (!L2) { fes.GetBoundaryTrueDofs(ess_dofs); }
146 
147  ParBilinearForm a(&fes);
148  if (H1 || L2)
149  {
150  a.AddDomainIntegrator(new MassIntegrator);
151  a.AddDomainIntegrator(new DiffusionIntegrator);
152  }
153  else
154  {
155  a.AddDomainIntegrator(new VectorFEMassIntegrator);
156  }
157 
158  if (ND) { a.AddDomainIntegrator(new CurlCurlIntegrator); }
159  else if (RT) { a.AddDomainIntegrator(new DivDivIntegrator); }
160  else if (L2)
161  {
162  a.AddInteriorFaceIntegrator(new DGDiffusionIntegrator(-1.0, kappa));
163  a.AddBdrFaceIntegrator(new DGDiffusionIntegrator(-1.0, kappa));
164  }
165  // TODO: L2 diffusion not implemented with partial assembly
166  if (!L2) { a.SetAssemblyLevel(AssemblyLevel::PARTIAL); }
167  a.Assemble();
168 
169  ParLinearForm b(&fes);
170  if (H1 || L2) { b.AddDomainIntegrator(new DomainLFIntegrator(f_coeff)); }
171  else { b.AddDomainIntegrator(new VectorFEDomainLFIntegrator(f_vec_coeff)); }
172  if (L2)
173  {
174  // DG boundary conditions are enforced weakly with this integrator.
175  b.AddBdrFaceIntegrator(new DGDirichletLFIntegrator(u_coeff, -1.0, kappa));
176  }
177  b.Assemble();
178 
179  ParGridFunction x(&fes);
180  if (H1 || L2) { x.ProjectCoefficient(u_coeff);}
181  else { x.ProjectCoefficient(u_vec_coeff); }
182 
183  Vector X, B;
184  OperatorHandle A;
185  a.FormLinearSystem(ess_dofs, x, b, A, X, B);
186 
187 
188  unique_ptr<Solver> solv_lor;
189  if (H1 || L2)
190  {
191  solv_lor.reset(new LORSolver<HypreBoomerAMG>(a, ess_dofs));
192  }
193  else if (RT && dim == 3)
194  {
195  solv_lor.reset(new LORSolver<HypreADS>(a, ess_dofs));
196  }
197  else
198  {
199  solv_lor.reset(new LORSolver<HypreAMS>(a, ess_dofs));
200  }
201 
202  CGSolver cg(MPI_COMM_WORLD);
203  cg.SetAbsTol(0.0);
204  cg.SetRelTol(1e-12);
205  cg.SetMaxIter(500);
206  cg.SetPrintLevel(1);
207  cg.SetOperator(*A);
208  cg.SetPreconditioner(*solv_lor);
209  cg.Mult(B, X);
210 
211  a.RecoverFEMSolution(X, b, x);
212 
213  double er =
214  (H1 || L2) ? x.ComputeL2Error(u_coeff) : x.ComputeL2Error(u_vec_coeff);
215  if (Mpi::Root()) { cout << "L2 error: " << er << endl; }
216 
217  if (visualization)
218  {
219  // Save the solution and mesh to disk. The output can be viewed using
220  // GLVis as follows: "glvis -np <np> -m mesh -g sol"
221  x.Save("sol");
222  mesh.Save("mesh");
223 
224  // Also save the solution for visualization using ParaView
225  ParaViewDataCollection dc("PLOR", &mesh);
226  dc.SetPrefixPath("ParaView");
227  dc.SetHighOrderOutput(true);
228  dc.SetLevelsOfDetail(order);
229  dc.RegisterField("u", &x);
230  dc.SetCycle(0);
231  dc.SetTime(0.0);
232  dc.Save();
233  }
234 
235  return 0;
236 }
