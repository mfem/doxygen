60  GridFunction &gf, const char *title,

62 {

64 
65  bool newly_opened = false;

67 
68  do

74  newly_opened = true;
75  }
76  sock << "solution\n";
77 
78  mesh.Print(sock);
79  gf.Save(sock);
80 
81  if (newly_opened)

83  sock << "window_title '" << title << "'\n"

86  << "keys maaAc";
87  if ( vec ) { sock << "vvv"; }
88  sock << endl;
89  }
90 
91  connection_failed = !sock && !newly_opened;
92  }
