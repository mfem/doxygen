34 inline void Sort3 (int &r, int &c, int &f)
35 {
36  int t;
37 
38  if (r > c)
39  if (c > f)
40  {
41  t = r; r = f; f = t; // (r,c,f) --> (f,c,r)
42  }
43  else if (r > f)
44  {
45  t = r; r = c; c = f; f = t; // (r,c,f) --> (c,f,r)
46  }
47  else
48  {
49  t = r; r = c; c = t; // (r,c,f) --> (c,r,f)
50  }

52  {
53  if (r > f)

59  t = c; c = f; f = t; // (r,c,f) --> (r,f,c)
