242 void Table::Finalize()
243 {
244  int i, j, end, sum = 0, n = 0, newI = 0;
245 
246  for (i=0; i<I[size]; i++)
247  {
248  if (J[i] != -1)
249  {
250  sum++;
251  }
252  }
253 
254  if (sum != I[size])
255  {
256  int *NewJ = Memory<int>(sum);
257 
258  for (i=0; i<size; i++)
259  {
260  end = I[i+1];
261  for (j=I[i]; j<end; j++)
262  {
263  if (J[j] == -1) { break; }
264  NewJ[ n++ ] = J[j];
265  }
266  I[i] = newI;
267  newI = n;
268  }
269  I[size] = sum;
270 
271  J.Delete();
272 
273  J.Wrap(NewJ, sum, true);
274 
275  MFEM_ASSERT(sum == n, "sum = " << sum << ", n = " << n);
276  }
