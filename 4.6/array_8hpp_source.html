474 /** A container for items of type T. Dynamically grows as items are added.
475  * Each item is accessible by its index. Items are allocated in larger chunks
476  * (blocks), so the 'Append' method is very fast on average.
477  */
478 template<typename T>
479 class BlockArray
480 {
481 public:
482  BlockArray(int block_size = 16*1024);

484  BlockArray& operator=(const BlockArray&) = delete; // not supported
485  ~BlockArray() { Destroy(); }

487  /// Allocate and construct a new item in the array, return its index.
488  int Append();
489 
490  /// Allocate and copy-construct a new item in the array, return its index.
491  int Append(const T &item);

493  /// Access item of the array.
494  inline T& At(int index)
495  {
496  CheckIndex(index);
497  return blocks[index >> shift][index & mask];
498  }
499  inline const T& At(int index) const
500  {
501  CheckIndex(index);

503  }

522 protected:
523  template <typename cA, typename cT>
524  class iterator_base
525  {
526  public:
527  cT& operator*() const { return *ptr; }
528  cT* operator->() const { return ptr; }
529 
530  bool good() const { return !stop; }
531  int index() const { return (ptr - ref); }
532 
533  protected:
534  cA *array;
535  cT *ptr, *b_end, *ref;
536  int b_end_idx;
537  bool stop;
538 
539  iterator_base() { }
540  iterator_base(bool stop) : stop(stop) { }
541  iterator_base(cA *a)
542  : array(a), ptr(a->blocks[0]), ref(ptr), stop(false)
543  {
544  b_end_idx = std::min(a->size, a->mask+1);
545  b_end = ptr + b_end_idx;
546  }
547 
548  void next()
549  {
550  MFEM_ASSERT(!stop, "invalid use");
551  if (++ptr == b_end)
552  {
553  if (b_end_idx < array->size)
554  {
555  ptr = &array->At(b_end_idx);
556  ref = ptr - b_end_idx;
557  b_end_idx = std::min(array->size, (b_end_idx|array->mask) + 1);
558  b_end = &array->At(b_end_idx-1) + 1;
559  }
560  else
561  {
562  MFEM_ASSERT(b_end_idx == array->size, "invalid use");
563  stop = true;
564  }
565  }
566  }
567  };
